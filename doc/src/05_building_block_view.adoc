== Bausteinsicht

=== Ebene 1: Gesamtsystem

[plantuml, target=level1-container, format=png]
----
@startuml
!include <C4/C4_Container>

Person(user, "User", "CTO/Tech Lead oder Mitarbeiter")

System_Boundary(system, "Weblab Tech Radar") {
  Container(client, "Client", "Angular, Nginx", "SPA mit Radar Viewer und Administration")
  Container(server, "Server", "Express, Prisma ORM", "REST API für Technologien")
  ContainerDb(db, "Datenbank", "PostgreSQL", "Technologie Einträge")
}

Rel(user, client, "", "")
Rel(client, server, "REST Aufrufe", "HTTP /api/technologies")
Rel(server, db, "liest/schreibt", "Prisma ORM")

@enduml
----

[cols="1,2" options="header"]
|===
| Baustein | Verantwortung
| Client | Angular SPA: Radar Viewer und Administration
| Server | REST API mit CRUD Operationen auf Technologien und Datenbankzugriff über das Prisma ORM
| Datenbank | Persistierung der Technologie Einträge
|===

=== Ebene 2: Client

[plantuml, target=level2-client, format=png]
----
@startuml
!include <C4/C4_Component>

Container_Boundary(client, "Client") {
  Component(viewer, "Tech Viewer", "Page", "Zeigt publizierte Technologien als Radar")
  Component(admin, "Tech Administration", "Page", "Technologien erfassen, ändern, publizieren")
  Component(techRadar, "Tech Radar", "Feature", "D3.js Tech-Radar Visualisierung")
  Component(techForm, "Tech Form", "Feature", "Formular zur Erfassung von Technologien")
  Component(techCards, "Tech Cards", "Feature", "Kartenansicht erfasster/publizierter Technologien")
  Component(apiService, "Tech API Service", "Service", "Kommunikation mit der REST API des Servers")
}

Rel(viewer, techRadar, "beinhaltet")
Rel(admin, techCards, "beinhaltet")
Rel(admin, techForm, "beinhaltet")
Rel(techRadar, apiService, "ruft auf")
Rel(techCards, apiService, "ruft auf")
Rel(techForm, apiService, "ruft auf")

@enduml
